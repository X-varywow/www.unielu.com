class Solution:
    def breakfastNumber(self, staple: List[int], drinks: List[int], x: int) -> int:
        mod=1000000007
        staple.sort()
        drinks.sort()
        ans=0
        for s in staple:
            a=bisect.bisect_right(drinks,x-s)
            if a:
                ans+=bisect.bisect_right(drinks,x-s)
            else: 
                break

        return ans%mod
