class Solution:
    def maxDepth(self, s: str) -> int:
        ans=0
        cnt=0
        for i in s:
            if i=="(": ans+=1
            elif i==")": ans-=1
            cnt=max(cnt,ans)
        return cnt

class Solution:
    def maximalNetworkRank(self, n: int, roads: List[List[int]]) -> int:
        ans=0
        d=defaultdict(list)
        for i,j in roads:
            d[i].append(j)
            d[j].append(i)
        for i in range(n):
            for j in range(i+1,n):
                tmp=len(d[i])+len(d[j])
                if i in d[j]: tmp-=1
                ans=max(ans,tmp)
        return ans

class Solution:
    def checkPalindromeFormation(self, a: str, b: str) -> bool:

            right=len(str1)-1-left
            while(left>=0 and right<len(str1)):
                if str1[left]!=str2[right]: break
                left-=1
                right+=1

class Solution:
    def countSubgraphsForEachDiameter(self, n, edges):
        d=defaultdict(list)
        for i, j in edges:
            d[i - 1].append(j - 1)
            d[j - 1].append(i - 1)

        def maxd(mask0):
            res = 0
            for i in range(n):
                if (1 << i) & mask0:
                    mask = mask0
                    bfs, bfs2 = [i], []
                    cur = 0
                    while bfs:
                        for i in bfs:
                            mask ^= 1 << i
                            for j in d[i]:
                                if mask & (1 << j):
                                    bfs2.append(j)
                        cur += 1
                        bfs, bfs2 = bfs2, []
                    if mask: return 0
                    res = max(res, cur - 1)
            return res

        res = [0] * (n - 1)
        for mask in range(1 << n):
            if mask & (mask - 1) == 0: continue
            k = maxd(mask)
            if k:
                res[k - 1] += 1

        return res
