import requests

def main(name, password, domain_name="", region="cn-north-4"):
    if domain_name == "":
        domain_name = name
    urls = "https://iam." + region + ".myhuaweicloud.com/v3/auth/tokens"
    header = {
        "User-Agent": "PostmanRuntime/7.24.1",
        "Content-Type": "application/json;charset=utf8",
    }
    json_data = {
        "auth": {
            "identity": {
                "methods": ["password"],
                "password": {
                    "user": {
                        "name": name,
                        "password": password,
                        "domain": {
                            "name": domain_name,
                        }
                    }
                }
            },
            "scope": {
                "project": {
                    "name": region
                }
            }
        }
    }
    z = requests.post(
                    url = urls,
                    headers = header,
                    json = json_data,
                    verify = False,
                    )

    if z.status_code == 201:
        return z.headers["X-Subject-Token"]
    elif z.status_code == 401:
        print(z.text)
        return ''
    else:
        print(z.status_code)
        return ''
